# Derivadas parciales para regresin lineal
# ####################################################################
def der_parcial_1(xs: list, ys: list) -> tuple[float, float, float]:
    """Retorna los coeficientes de la ecuacin de la derivada parcial con respecto al parmetro 1 al reemplazar los valores ``xs`` y ``ys``. La ecuacin es de la forma:
    c_1 * a_1 + c_0 * a_0 = c_ind


    ## Parameters

    ``xs``: lista de valores de x.

    ``ys``: lista de valores de y.

    ## Return

    ``c_1``: coeficiente del parmetro 1.

    ``c_0``: coeficiente del parmetro 0.

    ``c_ind``: coeficiente del trmino independiente.

    """

    # coeficiente del trmino independiente
    c_ind = sum(ys)

    # coeficiente del parmetro 1
    c_1 = sum(xs)

    # coeficiente del parmetro 0
    c_0 = len(xs)

    return (c_1, c_0, c_ind)


def der_parcial_0(xs: list, ys: list) -> tuple[float, float, float]:
    """Retorna los coeficientes de la ecuacin de la derivada parcial con respecto al parmetro 0 al reemplazar los valores ``xs`` y ``ys``. La ecuacin es de la forma:
    c_1 * a_1 + c_0 * a_0 = c_ind


    ## Parameters

    ``xs``: lista de valores de x.

    ``ys``: lista de valores de y.

    ## Return

    ``c_1``: coeficiente del parmetro 1.

    ``c_0``: coeficiente del parmetro 0.

    ``c_ind``: coeficiente del trmino independiente.

    """
    c_1 = 0
    c_0 = 0
    c_ind = 0
    for xi, yi in zip(xs, ys):
        # coeficiente del trmino independiente
        c_ind += xi * yi

        # coeficiente del parmetro 1
        c_1 += xi * xi

        # coeficiente del parmetro 0
        c_0 += xi

    return (c_1, c_0, c_ind)

    c_2 = sum(xi**3 for xi in xs)
    c_1 = sum(xi**2 for xi in xs)
    c_0 = sum(xs)

import numpy as np
# Definicin de la funcin cuadrtica
def cuadratica(x, pars):
    a2, a1, a0 = pars
    return a2 * x**2 + a1 * x + a0
# Implementacin para ajustar mnimos cuadrados
def ajustar_min_cuadrados(xs, ys):
    # Matriz A para el ajuste cuadrtico
    A = np.vstack([np.square(xs), xs, np.ones(len(xs))]).T
    pars, _, _, _ = np.linalg.lstsq(A, ys, rcond=None)
    return pars
# Ajustar los datos
pars = ajustar_min_cuadrados(xs1, ys1)
# Coeficientes del ajuste cuadrtico
a2, a1, a0 = pars

# Calcular y(2.25) y y(-2.25) usando la funcin cuadrtica ajustada
y_2_25 = cuadratica(2.25, pars)
y_neg_2_25 = cuadratica(-2.25, pars)

# Mostrar resultados
print(f"a2 = {a2}")
print(f"a1 = {a1}")
print(f"a0 = {a0}")
print(f"y(2.25) = {y_2_25}")

import numpy as np
import matplotlib.pyplot as plt

# Definicin de la funcin exponencial
def exponencial(x, pars):
    a, b = pars
    return a * np.exp(b * x)

# Ajustar mnimos cuadrados para funcin exponencial
def ajustar_exponencial(xs, ys):
    # Linealizar el problema: ln(y) = ln(a) + b * x
    log_ys = np.log(ys)
    A = np.vstack([xs, np.ones(len(xs))]).T
    b, log_a = np.linalg.lstsq(A, log_ys, rcond=None)[0]
    a = np.exp(log_a)
    return a, b

# Ajustar los datos al modelo exponencial (xs y ys definidos externamente)
pars = ajustar_exponencial(xs2, ys2)

# Generar puntos para la curva ajustada
