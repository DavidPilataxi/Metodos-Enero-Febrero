import sympy as sp

def lu_decomposition(mat, decimals=4):
    """Realiza la descomposicin LU con l_ii = 1 para todas las i y trunca a 'decimals' decimales."""
    size = mat.shape[0]
    lower = sp.eye(size)
    upper = mat.copy()

    for row in range(size):
        for col in range(row + 1, size):
            multiplier = upper[col, row] / upper[row, row]
            lower[col, row] = multiplier
            upper[col, :] -= multiplier * upper[row, :]

    # Redondear matrices
    lower = lower.applyfunc(lambda x: round(x, decimals))
    upper = upper.applyfunc(lambda x: round(x, decimals))

    return lower, upper

import numpy as np

def lu_decompose(matrix):
    """Descomposicin LU con Lii = 1 para todas las i."""
    n = len(matrix)
    L = np.eye(n)  # Matriz L con diagonal de unos
    U = np.zeros((n, n))

    for i in range(n):
        for j in range(i, n):
            U[i, j] = matrix[i, j] - sum(L[i, k] * U[k, j] for k in range(i))
        
        for j in range(i + 1, n):
            L[j, i] = (matrix[j, i] - sum(L[j, k] * U[k, i] for k in range(i))) / U[i, i]

    return L, U

def forward_subst(L, b):
    """Sustitucin progresiva para resolver Ly = b."""
    n = len(b)
    y = np.zeros(n)

    for i in range(n):
        y[i] = b[i] - sum(L[i, j] * y[j] for j in range(i))

    return y

def backward_subst(U, y):
    """Sustitucin regresiva para resolver Ux = y."""
    n = len(y)
    x = np.zeros(n)

    for i in range(n - 1, -1, -1):
        x[i] = (y[i] - sum(U[i, j] * x[j] for j in range(i + 1, n))) / U[i, i]

    return x

def solve_system(A, b):
    """Resuelve el sistema Ax = b usando descomposicin LU."""
    L, U = lu_decompose(A)
    y = forward_subst(L, b)
    x = backward_subst(U, y)
    return np.round(x, 4)  # Redondear la solucin a 4 decimales
